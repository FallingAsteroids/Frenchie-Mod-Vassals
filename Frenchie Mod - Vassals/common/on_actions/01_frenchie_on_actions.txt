# country
on_colonial_liberation = {
	if = {
		limit = {
			NOT = {
				OR = {
					has_country_flag = estate
					is_subject = yes
				}
			}
		}
		country_event = { id = colonial_nation.90 days = 1 }
	}
}

on_startup = {
	set_variable = {
		which = frenchie_estate_added_tax_income
		value = 0
	}
	set_variable = {
		which = frenchie_used_estate_added_tax_income
		value = 0
	}
}

on_annexed = {
	release = FROM
	create_subject = {
		subject_type = estate
		subject = FROM
	}
	FROM = {
		every_owned_province = {
			limit = {
				NOT = {
					any_province_in_state = {
						is_capital_of = FROM
					}
				}
			}
			cede_province = ROOT
		}
	}
}
